515 uint32_t CLK_EnablePLL(uint32_t u32PllClkSrc, uint32_t u32PllFreq)
516 {
517  uint32_t u32PllCr,u32PLL_N,u32PLL_M,u32PLLReg;
518  if ( u32PllFreq < FREQ_16MHZ)
519  u32PllFreq=FREQ_16MHZ;
520  else if(u32PllFreq > FREQ_32MHZ)
521  u32PllFreq=FREQ_32MHZ;
522 
523  if(u32PllClkSrc == CLK_PLLCTL_PLL_SRC_HXT)
524  {
525  /* PLL source clock from HXT */
526  CLK->PLLCTL &= ~CLK_PLLCTL_PLL_SRC_HIRC;
527  u32PllCr = __HXT;
528  }
529  else
530  {
531  /* PLL source clock from HIRC */
532  CLK->PLLCTL |= CLK_PLLCTL_PLL_SRC_HIRC;
533  if(CLK->PWRCTL & CLK_PWRCTL_HIRC_FSEL_Msk)
534  u32PllCr =__HIRC16M;
535  else
536  u32PllCr =__HIRC12M;
537  }
538 
539 
540  u32PLL_N=u32PllCr/1000000;
541  u32PLL_M=u32PllFreq/1000000;
542  while(1)
543  {
544  if(u32PLL_M<=32 && u32PLL_N<=16 ) break;
545  u32PLL_M >>=1;
546  u32PLL_N >>=1;
547  }
548  u32PLLReg = (u32PLL_M<<CLK_PLLCTL_PLL_MLP_Pos) | ((u32PLL_N-1)<<CLK_PLLCTL_PLL_SRC_N_Pos);
549  CLK->PLLCTL = ( CLK->PLLCTL & ~(CLK_PLLCTL_PLL_MLP_Msk | CLK_PLLCTL_PLL_SRC_N_Msk ) )| u32PLLReg;
550 
551  if(u32PllClkSrc==CLK_PLLCTL_PLL_SRC_HIRC)
552  CLK->PLLCTL = (CLK->PLLCTL & ~CLK_PLLCTL_PLL_SRC_HIRC) | (CLK_PLLCTL_PLL_SRC_HIRC);
553  else
554  CLK->PLLCTL = (CLK->PLLCTL & ~CLK_PLLCTL_PLL_SRC_HIRC);
555 
556  CLK->PLLCTL &= ~CLK_PLLCTL_PD_Msk;
557 
558  return CLK_GetPLLClockFreq();
559 }
